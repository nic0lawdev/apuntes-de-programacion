Operadores de Asignación

let z = 10;
z += 5; // Equivale a z = z + 5; (z = 15)
z -= 3; // Equivale a z = z - 3; (z = 12)
z *= 2; // Equivale a z = z * 2; (z = 24)
z /= 4; // Equivale a z = z / 4; (z = 6)
z %= 2; // Equivale a z = z % 2; (z = 0)
